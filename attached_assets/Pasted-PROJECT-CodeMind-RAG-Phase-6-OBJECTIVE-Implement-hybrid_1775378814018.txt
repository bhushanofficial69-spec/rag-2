5. **Implement Keyword Search Logic**
```python
   def keyword_search(self, query: str, top_k: int = 10):
       # Extract keywords from query
       keywords = self._extract_keywords(query)
       
       # Search Pinecone metadata for keyword matches
       # Option A: Simple substring match in code_snippet
       results = []
       for keyword in keywords:
           matched = self.vector_db.query(
               vector=None,  # Don't use vector
               filters={"code_snippet": {"$contains": keyword}},
               top_k=top_k
           )
           results.extend(matched)
       
       # Deduplicate and rank by keyword count
       unique_results = {r['id']: r for r in results}
       ranked = sorted(
           unique_results.values(),
           key=lambda x: x['keyword_match_count'],
           reverse=True
       )
       return ranked[:top_k]
```

6. **Implement RRF Merging**
```python
   def reciprocal_rank_fusion(self, semantic_results, sparse_results, k=60):
       # Assign RRF scores
       rrf_scores = {}
       
       for rank, result in enumerate(semantic_results, 1):
           score = 1 / (k + rank)
           if result['id'] not in rrf_scores:
               rrf_scores[result['id']] = {'result': result, 'score': 0}
           rrf_scores[result['id']]['score'] += score
       
       for rank, result in enumerate(sparse_results, 1):
           score = 1 / (k + rank)
           if result['id'] not in rrf_scores:
               rrf_scores[result['id']] = {'result': result, 'score': 0}
           rrf_scores[result['id']]['score'] += score
       
       # Sort by combined RRF score
       merged = sorted(
           rrf_scores.values(),
           key=lambda x: x['score'],
           reverse=True
       )
       
       return [item['result'] for item in merged]
```

7. **Add to routers/search.py**

```python
   @router.post("/search")
   async def search(request: SearchQuery):
       try:
           # Validate
           if not request.query.strip():
               return {"error": "Query cannot be empty"}, 400
           if request.top_k > 20:
               request.top_k = 20
           
           # Search
           start_time = time.time()
           results = hybrid_search_service.hybrid_search(
               query=request.query,
               top_k=request.top_k,
               filters=request.filters
           )
           search_time_ms = (time.time() - start_time) * 1000
           
           # Format response
           return HybridSearchResponse(
               query=request.query,
               results=results,
               total_results=len(results),
               search_time_ms=int(search_time_ms)
           )
       except Exception as e:
           logger.error(f"Search error: {e}")
           return {"error": str(e)}, 500
```
